819  std::pair<iterator, bool> insert_at(iterator i_element, value_type& value)
820  {
821  std::pair<iterator, bool> result(end(), false);
822 
823  if (i_element == end())
824  {
825  // At the end.
826  ETL_ASSERT(!lookup.full(), ETL_ERROR(flat_map_full));
827 
828  lookup.push_back(&value);
829  result.first = --end();
830  result.second = true;
831  }
832  else
833  {
834  // Not at the end.
835  result.first = i_element;
836 
837  // Existing element?
838  if (value.first != i_element->first)
839  {
840  // A new one.
841  ETL_ASSERT(!lookup.full(), ETL_ERROR(flat_map_full));
842  lookup.insert(i_element.ilookup, &value);
843  result.second = true;
844  }
845  }
846 
847  return result;
848  }
