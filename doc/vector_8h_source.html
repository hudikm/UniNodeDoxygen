221  //*********************************************************************
222  void resize(size_t new_size, T value)
223  {
224  ETL_ASSERT(new_size <= CAPACITY, ETL_ERROR(vector_full));
225 
226  const size_t current_size = size();
227  size_t delta = (current_size < new_size) ? new_size - current_size : current_size - new_size;
228 
229  if (current_size < new_size)
230  {
231  etl::uninitialized_fill_n(p_end, delta, value);
232 #if defined(ETL_DEBUG)
233  construct_count += delta;
234 #endif
235  }
236  else
237  {
238  etl::destroy_n(p_end - delta, delta);
239 #if defined(ETL_DEBUG)
240  construct_count -= delta;
241 #endif
242  }
243 
244  p_end = p_buffer + new_size;
245  }

649  //*********************************************************************
650  void insert(iterator position, size_t n, parameter_t value)
651  {
652  ETL_ASSERT((size() + n) <= CAPACITY, ETL_ERROR(vector_full));
653 
654  size_t insert_n = n;
655  size_t insert_begin = std::distance(begin(), position);
656  size_t insert_end = insert_begin + insert_n;
657 
658  // Copy old data.
659  size_t copy_old_n;
660  size_t construct_old_n;
661  iterator p_construct_old;
662 
663  if (insert_end > size())
664  {
665  copy_old_n = 0;
666  construct_old_n = size() - insert_begin;
667  p_construct_old = p_buffer + insert_end;
668  }
669  else
670  {
671  copy_old_n = size() - insert_begin - insert_n;
672  construct_old_n = insert_n;
673  p_construct_old = p_end;
674  }
675 
676  size_t copy_new_n = construct_old_n;
677  size_t construct_new_n = insert_n - copy_new_n;
678 
679 #if defined(ETL_DEBUG)
680  // Construct old.
681  etl::uninitialized_copy_n(p_end - construct_old_n, construct_old_n, p_construct_old, construct_count);
682 
683  // Copy old.
684  etl::copy_n(p_buffer + insert_begin, copy_old_n, p_buffer + insert_end);
685 
686  // Construct new.
687  etl::uninitialized_fill_n(p_end, construct_new_n, value, construct_count);
688 
689  // Copy new.
690  std::fill_n(p_buffer + insert_begin, copy_new_n, value);
691 #else
692  // Construct old.
693  etl::uninitialized_copy_n(p_end - construct_old_n, construct_old_n, p_construct_old);
694 
695  // Copy old.
696  etl::copy_n(p_buffer + insert_begin, copy_old_n, p_buffer + insert_end);
697 
698  // Construct new.
699  etl::uninitialized_fill_n(p_end, construct_new_n, value);
700 
701  // Copy new.
702  std::fill_n(p_buffer + insert_begin, copy_new_n, value);
703 #endif
704 
705  p_end += n;
706  }
