Utility class for creating an using a singleton and with use of factory method to create new instance

Note
Synchronization level: Thread safe
: This class must only be used in a static context - this class must never be allocated or created on the stack.
: This class is lazily initialized on first use. This class is a POD type so if it is not used it will be garbage collected.

auto factory_fn = [](uint32_t mem)-> Queue {return new (mem) Queue(10,sizeof(int));}; SingletonPtrFactoryFn<Queue> CommTask::skuska(factory_fn); //Or write lambda function directly SingletonPtr2<Queue> CommTask::skuska([](uint32_t mem)-> Queue {return new (mem) Queue(10,sizeof(int));});

#ifndef PLATFORM_SINGLETONPTR_H_
#define PLATFORM_SINGLETONPTR_H_
#include <assert.h>
#include "FreeRtosSemaphore.h"
#include <new>
#if (FREERTOS_USED)
extern FreeRtosSemaphore singleton_mutex_id;
#endif
inline static void singleton_lock(void) {
#if (FREERTOS_USED)
singleton_mutex_id.lock();
#endif
}
inline static void singleton_unlock(void) {
#if (FREERTOS_USED)
singleton_mutex_id.unlock();
#endif
}
template<class T>
struct SingletonPtr {
T* get() {
if (NULL == _ptr) {
singleton_lock();
if (NULL == _ptr) {
_ptr = new (_data) T();
}
singleton_unlock();
}
// _ptr was not zero initialized or was
// corrupted if this assert is hit
assert(_ptr == (T * ) &_data);
return _ptr;
}
T* operator->() {
return get();
}
// This is zero initialized when in global scope
T *_ptr;
// Force data to be 4 byte aligned
uint32_t _data[(sizeof(T) + sizeof(uint32_t) - 1) / sizeof(uint32_t)];
};
template<class T>
struct SingletonPtrFactoryFn {
int i;
typedef T* (*factory_fn_t)(uint32_t *);
factory_fn_t factory_fn = NULL;
SingletonPtrFactoryFn(factory_fn_t fn) :
factory_fn(fn) {
}
T* get() {
if (NULL == _ptr) {
singleton_lock();
if (NULL == _ptr) {
assert(
factory_fn == NULL || "Wrong factory method in SingletonPtr");
_ptr = factory_fn(_data);
}
singleton_unlock();
}
// _ptr was not zero initialized or was
// corrupted if this assert is hit
assert(_ptr == (T * ) &_data);
return _ptr;
}
T* operator->() {
return get();
}
// This is zero initialized when in global scope
T *_ptr;
// Force data to be 4 byte aligned
uint32_t _data[(sizeof(T) + sizeof(uint32_t) - 1) / sizeof(uint32_t)];
};
#endif /* PLATFORM_SINGLETONPTR_H_ */
