634  mapped_type& operator [](key_parameter_t key)
635  {
636  // Find the bucket.
637  bucket_t* pbucket = pbuckets + get_bucket_index(key);
638 
639  // Find the first node in the bucket.
640  local_iterator inode = pbucket->begin();
641

643  while (inode != pbucket->end())
644  {
645  // Equal keys?
646  if (key_equal_function(key, inode->key_value_pair.first))
647  {
648  // Found a match.
649  return inode->key_value_pair.second;

653  ++inode;

762  //*********************************************************************
766  //*********************************************************************
767  std::pair<iterator, bool> insert(const value_type& key_value_pair)
768  {
769  std::pair<iterator, bool> result(end(), false);
770 
771  ETL_ASSERT(!full(), ETL_ERROR(unordered_map_full));
772 
773  const key_type& key = key_value_pair.first;
774  const mapped_type& mapped = key_value_pair.second;
775 
776  // Get the hash index.
777  size_t index = get_bucket_index(key);
778 
779  // Get the bucket & bucket iterator.
780  bucket_t* pbucket = pbuckets + index;
781  bucket_t& bucket = *pbucket;
782 
783  size_t s = pbuckets->size();
784 
785  // The first one in the bucket?
786  if (bucket.empty())
787  {
788  // Get a new node.
789  node_t& node = *pnodepool->allocate<node_t>();
790  ::new (&node.key_value_pair) value_type(key_value_pair);
791  ++construct_count;
792 
793  // Just add the pointer to the bucket;
794  bucket.insert_after(bucket.before_begin(), node);
795 
796  result.first = iterator((pbuckets + number_of_buckets), pbucket, pbucket->begin());
797  result.second = true;
798 
799  adjust_first_last_markers(pbucket);
800  }
801  else
802  {
803  // Step though the bucket looking for a place to insert.
804  local_iterator inode_previous = bucket.before_begin();
805  local_iterator inode = bucket.begin();
806 
807  while (inode != bucket.end())
808  {
809  // Do we already have this key?
810  if (inode->key_value_pair.first == key)
811  {
812  break;
813  }
814 
815  ++inode_previous;
816  ++inode;
817  }
818 
819  // Not already there?
820  if (inode == bucket.end())
821  {
822  // Get a new node.
823  node_t& node = *pnodepool->allocate<node_t>();
824  ::new (&node.key_value_pair) value_type(key_value_pair);
825  ++construct_count;
826 
827  // Add the node to the end of the bucket;
828  bucket.insert_after(inode_previous, node);
829  ++inode_previous;
830 
831  result.first = iterator((pbuckets + number_of_buckets), pbucket, inode_previous);
832  result.second = true;
833  }
834  }
835 
836  return result;
837  }

871  size_t erase(key_parameter_t key)
872  {
873  size_t n = 0;
874  size_t index = get_bucket_index(key);
875 
876  bucket_t& bucket = pbuckets[index];
877 
878  local_iterator iprevious = bucket.before_begin();
879  local_iterator icurrent = bucket.begin();
880 
881  // Search for the key, if we have it.
882  while ((icurrent != bucket.end()) && (icurrent->key_value_pair.first != key))
883  {
884  ++iprevious;
885  ++icurrent;
886  }
887 
888  // Did we find it?
889  if (icurrent != bucket.end())
890  {
891  bucket.erase_after(iprevious); // Unlink from the bucket.
892  icurrent->key_value_pair.~value_type(); // Destroy the value.
893  pnodepool->release(&*icurrent); // Release it back to the pool.
894  n = 1;
895  --construct_count;
896  }
897 
898  return n;
899  }
