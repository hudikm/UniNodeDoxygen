787  //*************************************************************************
791  //*************************************************************************
792  template <typename TCompare>
793  void sort(TCompare compare)
794  {
795  iterator i_left;
796  iterator i_right;
797  iterator i_node;
798  iterator i_head;
799  iterator i_tail;
800  int list_size = 1;
801  int number_of_merges;
802  int left_size;
803  int right_size;
804 
805  if (this->is_trivial_list())
806  {
807  return;
808  }
809 
810  while (true)
811  {
812  i_left = begin();
813  i_head = end();
814  i_tail = end();
815 
816  number_of_merges = 0; // Count the number of merges we do in this pass.
817 
818  while (i_left != end())
819  {
820  ++number_of_merges; // There exists a merge to be done.
821  i_right = i_left;
822  left_size = 0;
823 
824  // Step 'list_size' places along from left
825  for (int i = 0; i < list_size; ++i)
826  {
827  ++left_size;
828  ++i_right;
829 
830  if (i_right == end())
831  {
832  break;
833  }
834  }
835 
836  // If right hasn't fallen off end, we have two lists to merge.
837  right_size = list_size;
838 
839  // Now we have two lists. Merge them.
840  while (left_size > 0 || (right_size > 0 && i_right != end()))
841  {
842  // Decide whether the next node of merge comes from left or right.
843  if (left_size == 0)
844  {
845  // Left is empty. The node must come from right.
846  i_node = i_right++;
847  --right_size;
848  }
849  else if (right_size == 0 || i_right == end())
850  {
851  // Right is empty. The node must come from left.
852  i_node = i_left++;
853  --left_size;
854  }
855  else if (!compare(*i_right, *i_left))
856  {
857  // First node of left is lower or same. The node must come from left.
858  i_node = i_left++;
859  --left_size;
860  }
861  else
862  {
863  // First node of right is lower. The node must come from right.
864  i_node = i_right;
865  ++i_right;
866  --right_size;
867  }
868 
869  // Add the next node to the merged head.
870  if (i_head == end())
871  {
872  etl::link<link_type>(i_head.p_value, i_node.p_value);
873  i_head = i_node;
874  i_tail = i_node;
875  }
876  else
877  {
878  etl::link<link_type>(i_tail.p_value, i_node.p_value);
879  i_tail = i_node;
880  }
881 
882  etl::link<link_type>(i_tail.p_value, this->terminal_link);
883  }
884 
885  // Now left has stepped `list_size' places along, and right has too.
886  i_left = i_right;
887  }
888 
889  // If we have done only one merge, we're finished.
890  if (number_of_merges <= 1) // Allow for number_of_merges == 0, the empty head case
891  {
892  return;
893  }
894 
895  // Otherwise repeat, merging lists twice the size
896  list_size *= 2;
897  }
898  }

944  void splice(iterator position, list_type& other)
945  {
946  // No point splicing to ourself!
947  if (&other != this)
948  {
949  if (!other.empty())
950  {
951  link_type& first = *other.get_head();
952  link_type& last = *other.get_tail();
953 
954  if (&other != this)
955  {
956  this->current_size += other.size();
957  }
958 
959  link_type& after = *position.p_value;
