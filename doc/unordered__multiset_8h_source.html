648  //*********************************************************************
652  //*********************************************************************
653  std::pair<iterator, bool> insert(const value_type& key)
654  {
655  std::pair<iterator, bool> result(end(), false);
656 
657  ETL_ASSERT(!full(), ETL_ERROR(unordered_multiset_full));
658 
659  // Get the hash index.
660  size_t index = get_bucket_index(key);
661 
662  // Get the bucket & bucket iterator.
663  bucket_t* pbucket = pbuckets + index;
664  bucket_t& bucket = *pbucket;
665 
666  // The first one in the bucket?
667  if (bucket.empty())
668  {
669  // Get a new node.
670  node_t& node = *pnodepool->allocate<node_t>();
671  ::new (&node.key) value_type(key);
672  ++construct_count;
673 
674  // Just add the pointer to the bucket;
675  bucket.insert_after(bucket.before_begin(), node);
676 
677  result.first = iterator((pbuckets + number_of_buckets), pbucket, pbucket->begin());
678  result.second = true;
679 
680  adjust_first_last_markers(pbucket);
681  }
682  else
683  {
684  // Step though the bucket looking for a place to insert.
685  local_iterator inode_previous = bucket.before_begin();
686  local_iterator inode = bucket.begin();
687 
688  while (inode != bucket.end())
689  {
690  // Do we already have this key?
691  if (inode->key == key)
692  {
693  break;
694  }
695 
696  ++inode_previous;
697  ++inode;
698  }
699 
700  // Get a new node.
701  node_t& node = *pnodepool->allocate<node_t>();
702  ::new (&node.key) value_type(key);
703  ++construct_count;
704 
705  // Add the node to the end of the bucket;
706  bucket.insert_after(inode_previous, node);
707  ++inode_previous;
708 
709  result.first = iterator((pbuckets + number_of_buckets), pbucket, inode_previous);
710  result.second = true;
711  }
712 
713  return result;
714  }

748  size_t erase(key_parameter_t key)
749  {
750  size_t n = 0;
751  size_t bucket_id = get_bucket_index(key);
752 
753  bucket_t& bucket = pbuckets[bucket_id];
754 
755  local_iterator iprevious = bucket.before_begin();
756  local_iterator icurrent = bucket.begin();
757 
758  while (icurrent != bucket.end())
759  {
760  if (icurrent->key == key)
761  {
762  bucket.erase_after(iprevious); // Unlink from the bucket.
763  icurrent->key.~value_type(); // Destroy the value.
764  pnodepool->release(&*icurrent); // Release it back to the pool.
765  ++n;
766  icurrent = iprevious;
767  --construct_count;
768  }
769  else
770  {
771  ++iprevious;
772  }
773 
774  ++icurrent;
775  }
776 
777  return n;
778  }

819 
820  // Get the starting point.
821  bucket_t* pbucket = first_.get_bucket_list_iterator();
822  local_iterator iprevious = pbucket->before_begin();
823  local_iterator icurrent = first_.get_local_iterator();
824  local_iterator iend = last_.get_local_iterator(); // Note: May not be in the same bucket as icurrent.
825 
826  // Find the node previous to the first one.
827  while (iprevious->etl_next != &*icurrent)
828  {
829  ++iprevious;
830  }
831 
832  while (icurrent != iend)
833  {
834 
835  local_iterator inext = pbucket->erase_after(iprevious); // Unlink from the bucket.
836  icurrent->key.~value_type(); // Destroy the value.
837  pnodepool->release(&*icurrent); // Release it back to the pool.
838  --construct_count;
839 
840  icurrent = inext;
841 
842  // Are we there yet?
843  if (icurrent != iend)
844  {
845  // At the end of this bucket?
846  if ((icurrent == pbucket->end()))
847  {
848  // Find the next non-empty one.
849  do

853 
854  iprevious = pbucket->before_begin();
855  icurrent = pbucket->begin();
856  }
857  }
858  }
