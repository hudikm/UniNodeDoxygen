491 static inline void SPI_SetPinMode(SPI_Type *base, spi_pin_mode_t pinMode)
492 {
493  /* Clear SPC0 and BIDIROE bit. */
494  base->C2 &= ~(SPI_C2_BIDIROE_MASK | SPI_C2_SPC0_MASK);
495  /* Set pin mode for transfer. */
496  base->C2 |= SPI_C2_BIDIROE(pinMode >> 1U) | SPI_C2_SPC0(pinMode & 1U);
497 }
