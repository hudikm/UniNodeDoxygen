257  //*************************************************************************
258  void detach_node(Node*& position, Node*& replacement)
259  {
260  // Make temporary copy of actual nodes involved because we might lose
261  // their references in the process (e.g. position is the same as
262  // replacement or replacement is a child of position)
263  Node* detached = position;
264  Node* swap = replacement;
265 
266  // Update current position to point to swap (replacement) node first
267  position = swap;
268 
269  // Update replacement node to point to child in opposite direction
270  // otherwise we might lose the other child of the swap node
271  replacement = swap->children[1 - swap->dir];
272 
273  // Point swap node to detached node's parent, children and weight
274  swap->parent = detached->parent;
275  swap->children[kLeft] = detached->children[kLeft];
276  swap->children[kRight] = detached->children[kRight];
277  if (swap->children[kLeft])
278  {
279  swap->children[kLeft]->parent = swap;
280  }
281  if (swap->children[kRight])
282  {
283  swap->children[kRight]->parent = swap;
284  }
285  swap->weight = detached->weight;
286  }

288  //*************************************************************************
290  //*************************************************************************
291  void balance_node(Node*& critical_node)
292  {
293  // Step 1: Update weights for all children of the critical node up to the
294  // newly inserted node. This step is costly (in terms of traversing nodes
295  // multiple times during insertion) but doesn't require as much recursion
296  Node* weight_node = critical_node->children[critical_node->dir];
297  while (weight_node)
298  {
299  // Keep going until we reach a terminal node (dir == kNeither)
300  if (kNeither != weight_node->dir)
301  {
302  // Does this insert balance the previous weight factor value?
303  if (weight_node->weight == 1 - weight_node->dir)
304  {
305  weight_node->weight = kNeither;
306  }
307  else
308  {
309  weight_node->weight = weight_node->dir;
310  }
311 
312  // Update weight factor node to point to next node
313  weight_node = weight_node->children[weight_node->dir];
314  }
315  else
316  {
317  // Stop loop, terminal node found
318  break;
319  }
320  } // while(weight_node)
321 
322  // Step 2: Update weight for critical_node or rotate tree to balance node
323  if (kNeither == critical_node->weight)
324  {
325  critical_node->weight = critical_node->dir;
326  }
327  // If direction is different than weight, then it will now be balanced
328  else if (critical_node->dir != critical_node->weight)
329  {
330  critical_node->weight = kNeither;
331  }
332  // Rotate is required to balance the tree at the critical node
333  else
334  {
335  // If critical node matches child node direction then perform a two
336  // node rotate in the direction of the critical node
337  if (critical_node->weight == critical_node->children[critical_node->dir]->dir)
338  {
339  rotate_2node(critical_node, critical_node->dir);
340  }
341  // Otherwise perform a three node rotation in the direction of the
342  // critical node
343  else
344  {
345  rotate_3node(critical_node, critical_node->dir,
346  critical_node->children[critical_node->dir]->children[1 - critical_node->dir]->dir);
347  }
348  }
349  }

508  //*************************************************************************
510  //*************************************************************************
511  void rotate_2node(Node*& position, uint_least8_t dir)
512  {
513  // A C A B
514  // B C -> A E OR B C -> D A
515  // D E B D D E E C
516  // C (new position) becomes the root
517  // A (position) takes ownership of D as its children[kRight] child
518  // C (new position) takes ownership of A as its left child
519  // OR
520  // B (new position) becomes the root
521  // A (position) takes ownership of E as its left child
522  // B (new position) takes ownership of A as its right child
523 
524  // Capture new root (either B or C depending on dir) and its parent
525  Node* new_root = position->children[dir];
526 
527  // Replace position's previous child with new root's other child
528  position->children[dir] = new_root->children[1 - dir];
529  // Update new root's other child parent pointer
530  if (position->children[dir])
531  {
532  position->children[dir]->parent = position;
533  }
534 
535  // New root's parent becomes current position's parent
536  new_root->parent = position->parent;
537  new_root->children[1 - dir] = position;
538  new_root->dir = 1 - dir;
539 
540  // Clear weight factor from current position
541  position->weight = kNeither;
542  // Position's parent becomes new_root
543  position->parent = new_root;
544  position = new_root;
545  // Clear weight factor from new root
546  position->weight = kNeither;
547  }
548 
549  //*************************************************************************
551  //*************************************************************************
552  void rotate_3node(Node*& position, uint_least8_t dir, uint_least8_t third)
553  {
554  // __A__ __E__ __A__ __D__
555  // _B_ C -> B A OR B _C_ -> A C
556  // D E D F G C D E B F G E
557  // F G F G
558  // E (new position) becomes the root
559  // B (position) takes ownership of F as its left child
560  // A takes ownership of G as its right child
561  // OR
562  // D (new position) becomes the root
563  // A (position) takes ownership of F as its right child
564  // C takes ownership of G as its left child
565 
566  // Capture new root (either E or D depending on dir)
567  Node* new_root = position->children[dir]->children[1 - dir];
568  // Set weight factor for B or C based on F or G existing and being a different than dir
569  position->children[dir]->weight = third != kNeither && third != dir ? dir : uint_least8_t(kNeither);
570 
571  // Detach new root from its tree (replace with new roots child)
572  position->children[dir]->children[1 - dir] = new_root->children[dir];
573  // Update new roots child parent pointer
574  if (new_root->children[dir])
575  {
576  new_root->children[dir]->parent = position->children[dir];
577  }
578 
579  // Attach current left tree to new root and update its parent
580  new_root->children[dir] = position->children[dir];
581  position->children[dir]->parent = new_root;
582 
583  // Set weight factor for A based on F or G
584  position->weight = third != kNeither && third == dir ? 1 - dir : kNeither;
585 
586  // Move new root's right tree to current roots left tree
587  position->children[dir] = new_root->children[1 - dir];
588  if (new_root->children[1 - dir])
589  {
590  new_root->children[1 - dir]->parent = position;
591  }
592 
593  // Attach current root to new roots right tree and assume its parent
594  new_root->parent = position->parent;
595  new_root->children[1 - dir] = position;
596  new_root->dir = 1 - dir;
597 
598  // Update current position's parent and replace with new root
599  position->parent = new_root;
600  position = new_root;
601  // Clear weight factor for new current position
602  position->weight = kNeither;
603  }

1535  Node* insert_node(Node*& position, Data_Node& node)
1536  {

1538  Node* found = position;
1539 
1540  // Was position provided not empty? then find where the node belongs
1541  if (position)
1542  {
1543  // Find the critical parent node (default to nullptr)
1544  Node* critical_parent_node = nullptr;
1545  Node* critical_node = root_node;
1546 
1547  while (found)
1548  {
1549  // Search for critical weight node (all nodes whose weight factor
1550  // is set to kNeither (balanced)
1551  if (kNeither != found->weight)
1552  {
1553  critical_node = found;
1554  }
1555 
1556  // Downcast found to Data_Node class for comparison and other operations
1557  Data_Node& found_data_node = imultiset::data_cast(*found);
1558 
1559  // Is the node provided to the left of the current position?
1560  if (node_comp(node, found_data_node))
1561  {
1562  // Update direction taken to insert new node in parent node
1563  found->dir = kLeft;
1564  }
1565  // Is the node provided to the right of the current position?
1566  else if (node_comp(found_data_node, node))
1567  {
1568  // Update direction taken to insert new node in parent node
1569  found->dir = kRight;
1570  }
1571  else
1572  {
1573  // Update direction taken to insert new node in parent (and
1574  // duplicate) node to the right.
1575  found->dir = kRight;
1576  }
1577 
1578  // Is there a child of this parent node?
1579  if (found->children[found->dir])
1580  {
1581  // Will this node be the parent of the next critical node whose
1582  // weight factor is set to kNeither (balanced)?
1583  if (kNeither != found->children[found->dir]->weight)
1584  {
1585  critical_parent_node = found;
1586  }
1587 
1588  // Keep looking for empty spot to insert new node
1589  found = found->children[found->dir];
1590  }
1591  else
1592  {
1593  // Attach node as a child of the parent node found
1594  attach_node(found, found->children[found->dir], node);
1595 
1596  // Return newly added node
1597  found = found->children[found->dir];
1598 
1599  // Exit loop
1600  break;
1601  }
1602  }
1603 
1604  // Was a critical node found that should be checked for balance?
1605  if (critical_node)
1606  {
1607  if (critical_parent_node == nullptr && critical_node == root_node)
1608  {
1609  balance_node(root_node);
1610  }
1611  else if (critical_parent_node == nullptr && critical_node == position)
1612  {
1613  balance_node(position);
1614  }
1615  else
1616  {
1617  balance_node(critical_parent_node->children[critical_parent_node->dir]);
1618  }
1619  }
1620  }
1621  else
1622  {
1623  // Attatch node to current position (which is assumed to be root)
1624  attach_node(nullptr, position, node);
1625 
1626  // Return newly added node at current position
1627  found = position;
1628  }
1629 
1630  // Return the node found (might be nullptr)
1631  return found;

1634  //*************************************************************************
1637  //*************************************************************************
1638  void remove_node(Node* node)
1639  {
1640  // If valid found node was provided then proceed with steps 1 through 5
1641  if (node)
1642  {
1643  // Downcast found node provided to Data_Node class
1644  Data_Node& data_node = imultiset::data_cast(*node);
1645 
1646  // Keep track of node as found node
1647  Node* found = node;
1648 
1649  // Step 1: Mark path from node provided back to the root node using the
1650  // internal temporary dir member value and using the parent pointer. This
1651  // will allow us to avoid recursion in finding the node in a tree that
1652  //might contain duplicate keys to be found.
1653  while (node)
1654  {
1655  if (node->parent)
1656  {
1657  // Which direction does parent use to get to this node?
1658  node->parent->dir =
1659  node->parent->children[kLeft] == node ? kLeft : kRight;
1660 
1661  // Make this nodes parent the next node
1662  node = node->parent;
1663  }
1664  else
1665  {
1666  // Root node found - break loop
1667  break;
1668  }
1669  }
1670 
1671  // Step 2: Follow the path provided above until we reach the node
1672  // provided and look for the balance node to start rebalancing the tree
1673  // from (up to the replacement node that will be found in step 3)
1674  Node* balance = root_node;
1675  while (node)
1676  {
1677  // Did we reach the node provided originally (found) then go to step 3
1678  if (node == found)
1679  {
1680  // Update the direction towards a replacement node at the found node
1681  node->dir = node->children[kLeft] ? kLeft : kRight;
1682 
1683  // Exit loop and proceed with step 3
1684  break;
1685  }
1686  else
1687  {
1688  // If this nodes weight is kNeither or we are taking the shorter path
1689  // to the next node and our sibling (on longer path) is balanced then
1690  // we need to update the balance node to this node but all our
1691  // ancestors will not require rebalancing
1692  if ((node->weight == kNeither) ||
1693  (node->weight == (1 - node->dir) &&
1694  node->children[1 - node->dir]->weight == kNeither))
1695  {
1696  // Update balance node to this node
1697  balance = node;
1698  }
1699 
1700  // Keep searching for found in the direction provided in step 1
1701  node = node->children[node->dir];
1702  }
1703  }
1704  // The value for node should not be nullptr at this point otherwise
1705  // step 1 failed to provide the correct path to found. Step 5 will fail
1706  // (probably subtly) if node should be nullptr at this point
1707 
1708  // Step 3: Find the node (node should be equal to found at this point)
1709  // to replace found with (might end up equal to found) while also
1710  // continuing to update balance the same as in step 2 above.
1711  while (node)
1712  {
1713  // Replacement node found if its missing a child in the replace->dir
1714  // value set at the end of step 2 above
1715  if (node->children[node->dir] == nullptr)
1716  {
1717  // Exit loop once node to replace found is determined
1718  break;
1719  }
1720 
1721  // If this nodes weight is kNeither or we are taking the shorter path
1722  // to the next node and our sibling (on longer path) is balanced then
1723  // we need to update the balance node to this node but all our
1724  // ancestors will not require rebalancing
1725  if ((node->weight == kNeither) ||
1726  (node->weight == (1 - node->dir) &&
1727  node->children[1 - node->dir]->weight == kNeither))
1728  {
1729  // Update balance node to this node
1730  balance = node;
1731  }
1732 
1733  // Keep searching for replacement node in the direction specified above
1734  node = node->children[node->dir];
1735 
1736  // Downcast node to Data_Node class for comparison operations
1737  Data_Node& replace_data_node = imultiset::data_cast(*node);
1738 
1739  // Compare the key provided to the replace data node key
1740  if (node_comp(data_node, replace_data_node))
1741  {
1742  // Update the direction to the replace node
1743  node->dir = kLeft;
1744  }
1745  else if (node_comp(replace_data_node, data_node))
1746  {
1747  // Update the direction to the replace node
1748  node->dir = kRight;
1749  }
1750  else
1751  {
1752  // Update the direction to the replace node
1753  node->dir = node->children[kLeft] ? kLeft : kRight;
1754  }
1755  } // while(node)
1756
