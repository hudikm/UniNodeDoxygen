439 static inline void I2C_MasterClearStatusFlags(I2C_Type *base, uint32_t statusMask)
440 {
441 /* Must clear the STARTF / STOPF bits prior to clearing IICIF */
442 #if defined(FSL_FEATURE_I2C_HAS_START_STOP_DETECT) && FSL_FEATURE_I2C_HAS_START_STOP_DETECT
443  if (statusMask & kI2C_StartDetectFlag)
444  {
445  /* Shift the odd-ball flags back into place. */
446  base->FLT |= (uint8_t)(statusMask >> 8U);
447  }
448 #endif
449 
450 #ifdef I2C_HAS_STOP_DETECT
451  if (statusMask & kI2C_StopDetectFlag)
452  {
453  /* Shift the odd-ball flags back into place. */
454  base->FLT |= (uint8_t)(statusMask >> 8U);
455  }
456 #endif
457 
458  base->S = (uint8_t)statusMask;
459 }
