264 /*------------------------------------------------------------------------------
265 
266  clock_gate_t definition:
267 
268  31 16 0
269  -----------------------------------------------------------------
270  | SIM_SCGC register offset | control bit offset in SCGC |
271  -----------------------------------------------------------------
272 
273  For example, the SDHC clock gate is controlled by SIM_SCGC3[17], the
274  SIM_SCGC3 offset in SIM is 0x1030, then kCLOCK_GateSdhc0 is defined as
275 
276  kCLOCK_GateSdhc0 = (0x1030 << 16) | 17;
277 
278 ------------------------------------------------------------------------------*/
279 
280 #define CLK_GATE_REG_OFFSET_SHIFT 16U
281 #define CLK_GATE_REG_OFFSET_MASK 0xFFFF0000U
282 #define CLK_GATE_BIT_SHIFT_SHIFT 0U
283 #define CLK_GATE_BIT_SHIFT_MASK 0x0000FFFFU
284 
285 #define CLK_GATE_DEFINE(reg_offset, bit_shift) \
286  ((((reg_offset) << CLK_GATE_REG_OFFSET_SHIFT) & CLK_GATE_REG_OFFSET_MASK) | \
287  (((bit_shift) << CLK_GATE_BIT_SHIFT_SHIFT) & CLK_GATE_BIT_SHIFT_MASK))
288 
289 #define CLK_GATE_ABSTRACT_REG_OFFSET(x) (((x)&CLK_GATE_REG_OFFSET_MASK) >> CLK_GATE_REG_OFFSET_SHIFT)
290 #define CLK_GATE_ABSTRACT_BITS_SHIFT(x) (((x)&CLK_GATE_BIT_SHIFT_MASK) >> CLK_GATE_BIT_SHIFT_SHIFT)

454 static inline void CLOCK_EnableClock(clock_ip_name_t name)
455 {
456  uint32_t regAddr = SIM_BASE + CLK_GATE_ABSTRACT_REG_OFFSET((uint32_t)name);
457  (*(volatile uint32_t *)regAddr) |= (1U << CLK_GATE_ABSTRACT_BITS_SHIFT((uint32_t)name));
458 }

465 static inline void CLOCK_DisableClock(clock_ip_name_t name)
466 {
467  uint32_t regAddr = SIM_BASE + CLK_GATE_ABSTRACT_REG_OFFSET((uint32_t)name);
468  (*(volatile uint32_t *)regAddr) &= ~(1U << CLK_GATE_ABSTRACT_BITS_SHIFT((uint32_t)name));
469 }
