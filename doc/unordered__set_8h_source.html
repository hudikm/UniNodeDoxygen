649  //*********************************************************************
653  //*********************************************************************
654  std::pair<iterator, bool> insert(const value_type& key)
655  {
656  std::pair<iterator, bool> result(end(), false);
657 
658  ETL_ASSERT(!full(), ETL_ERROR(unordered_set_full));
659 
660  // Get the hash index.
661  size_t index = get_bucket_index(key);
662 
663  // Get the bucket & bucket iterator.
664  bucket_t* pbucket = pbuckets + index;
665  bucket_t& bucket = *pbucket;
666 
667  // The first one in the bucket?
668  if (bucket.empty())
669  {
670  // Get a new node.
671  node_t& node = *pnodepool->allocate<node_t>();
672  ::new (&node.key) value_type(key);
673  ++construct_count;
674 
675  // Just add the pointer to the bucket;
676  bucket.insert_after(bucket.before_begin(), node);
677 
678  result.first = iterator(pbuckets + number_of_buckets, pbucket, pbucket->begin());
679  result.second = true;
680 
681  adjust_first_last_markers(pbucket);
682  }
683  else
684  {
685  // Step though the bucket looking for a place to insert.
686  local_iterator inode_previous = bucket.before_begin();
687  local_iterator inode = bucket.begin();
688 
689  while (inode != bucket.end())
690  {
691  // Do we already have this key?
692  if (inode->key == key)
693  {
694  break;
695  }
696 
697  ++inode_previous;
698  ++inode;
699  }
700 
701  // Not already there?
702  if (inode == bucket.end())
703  {
704  // Get a new node.
705  node_t& node = *pnodepool->allocate<node_t>();
706  ::new (&node.key) value_type(key);
707  ++construct_count;
708 
709  // Add the node to the end of the bucket;
710  bucket.insert_after(inode_previous, node);
711  ++inode_previous;
712 
713  result.first = iterator(pbuckets + number_of_buckets, pbucket, inode_previous);
714  result.second = true;
715  }
716  }
717 
718  return result;
719  }

753  size_t erase(key_parameter_t key)
754  {
755  size_t n = 0;
756  size_t index = get_bucket_index(key);
757 
758  bucket_t& bucket = pbuckets[index];
759 
760  local_iterator iprevious = bucket.before_begin();
761  local_iterator icurrent = bucket.begin();
762 
763  // Search for the key, if we have it.
764  while ((icurrent != bucket.end()) && (icurrent->key != key))
765  {
766  ++iprevious;
767  ++icurrent;
768  }
769 
770  // Did we find it?
771  if (icurrent != bucket.end())
772  {
773  bucket.erase_after(iprevious); // Unlink from the bucket.
774  icurrent->key.~value_type(); // Destroy the value.
775  pnodepool->release(&*icurrent); // Release it back to the pool.
776  n = 1;
777  --construct_count;
778  }
779 
780  return n;
781  }

825  local_iterator iprevious = pbucket->before_begin();
826  local_iterator icurrent = first_.get_local_iterator();
827  local_iterator iend = last_.get_local_iterator(); // Note: May not be in the same bucket as icurrent.
828 
829  // Find the node previous to the first one.

831  {
832  ++iprevious;
833  }
834 
835  while (icurrent != iend)
836  {
837 
838  local_iterator inext = pbucket->erase_after(iprevious); // Unlink from the bucket.
839  icurrent->key.~value_type(); // Destroy the value.
840  pnodepool->release(&*icurrent); // Release it back to the pool.
841  --construct_count;
842 
843  icurrent = inext;
844 
845  // Are we there yet?
846  if (icurrent != iend)
847  {
848  // At the end of this bucket?
