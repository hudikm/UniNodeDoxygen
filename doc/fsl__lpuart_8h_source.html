622 static inline uint8_t LPUART_ReadByte(LPUART_Type *base)
623 {
624 #if defined(FSL_FEATURE_LPUART_HAS_7BIT_DATA_SUPPORT) && FSL_FEATURE_LPUART_HAS_7BIT_DATA_SUPPORT
625  uint32_t ctrl = base->CTRL;
626  bool isSevenDataBits =
627  ((ctrl & LPUART_CTRL_M7_MASK) ||
628  ((!(ctrl & LPUART_CTRL_M7_MASK)) && (!(ctrl & LPUART_CTRL_M_MASK)) && (ctrl & LPUART_CTRL_PE_MASK)));
629 
630  if (isSevenDataBits)
631  {
632  return (base->DATA & 0x7F);
633  }
634  else
635  {
636  return base->DATA;
637  }
638 #else
639  return base->DATA;
640 #endif
641 }
