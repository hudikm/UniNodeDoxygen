246  void balance_node(Node*& critical_node)
247  {
248  // Step 1: Update weights for all children of the critical node up to the
249  // newly inserted node. This step is costly (in terms of traversing nodes
250  // multiple times during insertion) but doesn't require as much recursion
251  Node* weight_node = critical_node->children[critical_node->dir];
252  while (weight_node)
253  {
254  // Keep going until we reach a terminal node (dir == kNeither)
255  if (uint_least8_t(kNeither) != weight_node->dir)
256  {
257  // Does this insert balance the previous weight factor value?
258  if (weight_node->weight == 1 - weight_node->dir)
259  {
260  weight_node->weight = uint_least8_t(kNeither);
261  }
262  else
263  {
264  weight_node->weight = weight_node->dir;
265  }
266 
267  // Update weight factor node to point to next node
268  weight_node = weight_node->children[weight_node->dir];
269  }
270  else
271  {
272  // Stop loop, terminal node found
273  break;
274  }
275  } // while(weight_node)
276 
277  // Step 2: Update weight for critical_node or rotate tree to balance node
278  if (uint_least8_t(kNeither) == critical_node->weight)
279  {
280  critical_node->weight = critical_node->dir;
281  }
282  // If direction is different than weight, then it will now be balanced
283  else if (critical_node->dir != critical_node->weight)
284  {
285  critical_node->weight = uint_least8_t(kNeither);
286  }
287  // Rotate is required to balance the tree at the critical node
288  else
289  {
290  // If critical node matches child node direction then perform a two
291  // node rotate in the direction of the critical node
292  if (critical_node->weight == critical_node->children[critical_node->dir]->dir)
293  {
294  rotate_2node(critical_node, critical_node->dir);
295  }
296  // Otherwise perform a three node rotation in the direction of the
297  // critical node
298  else
299  {
300  rotate_3node(critical_node, critical_node->dir,
301  critical_node->children[critical_node->dir]->children[1 - critical_node->dir]->dir);
302  }
303  }
304  }

306  //*************************************************************************
308  //*************************************************************************
309  void rotate_2node(Node*& position, uint_least8_t dir)
310  {
311  // A C A B
312  // B C -> A E OR B C -> D A
313  // D E B D D E E C
314  // C (new position) becomes the root
315  // A (position) takes ownership of D as its children[kRight] child
316  // C (new position) takes ownership of A as its left child
317  // OR
318  // B (new position) becomes the root
319  // A (position) takes ownership of E as its left child
320  // B (new position) takes ownership of A as its right child
321 
322  // Capture new root
323  Node* new_root = position->children[dir];
324  // Replace position's previous child with new root's other child
325  position->children[dir] = new_root->children[1 - dir];
326  // New root now becomes parent of current position
327  new_root->children[1 - dir] = position;
328  // Clear weight factor from current position
329  position->weight = uint_least8_t(kNeither);
330  // Newly detached right now becomes current position
331  position = new_root;
332  // Clear weight factor from new root
333  position->weight = uint_least8_t(kNeither);
334  }
335 
336  //*************************************************************************
338  //*************************************************************************
339  void rotate_3node(Node*& position, uint_least8_t dir, uint_least8_t third)
340  {
341  // __A__ __E__ __A__ __D__
342  // _B_ C -> B A OR B _C_ -> A C
343  // D E D F G C D E B F G E
344  // F G F G
345  // E (new position) becomes the root
346  // B (position) takes ownership of F as its left child
347  // A takes ownership of G as its right child
348  // OR
349  // D (new position) becomes the root
350  // A (position) takes ownership of F as its right child
351  // C takes ownership of G as its left child
352 
353  // Capture new root (either E or D depending on dir)
354  Node* new_root = position->children[dir]->children[1 - dir];
355  // Set weight factor for B or C based on F or G existing and being a different than dir
356  position->children[dir]->weight = third != uint_least8_t(kNeither) && third != dir ? dir : uint_least8_t(kNeither);
357 
358  // Detach new root from its tree (replace with new roots child)
359  position->children[dir]->children[1 - dir] =
360  new_root->children[dir];
361  // Attach current left tree to new root
362  new_root->children[dir] = position->children[dir];
363  // Set weight factor for A based on F or G
364  position->weight = third != uint_least8_t(kNeither) && third == dir ? 1 - dir : uint_least8_t(kNeither);
365 
366  // Move new root's right tree to current roots left tree
367  position->children[dir] = new_root->children[1 - dir];
368  // Attach current root to new roots right tree
369  new_root->children[1 - dir] = position;
370  // Replace current position with new root
371  position = new_root;
372  // Clear weight factor for new current position
373  position->weight = uint_least8_t(kNeither);
374  }

428  void detach_node(Node*& position, Node*& replacement)
429  {
430  // Make temporary copy of actual nodes involved because we might lose
431  // their references in the process (e.g. position is the same as
432  // replacement or replacement is a child of position)
433  Node* detached = position;
434  Node* swap = replacement;
435 
436  // Update current position to point to swap (replacement) node first
437  position = swap;
438 
439  // Update replacement node to point to child in opposite direction
440  // otherwise we might lose the other child of the swap node
441  replacement = swap->children[1 - swap->dir];
442 
443  // Point swap node to detached node's children and weight
444  swap->children[kLeft] = detached->children[kLeft];
445  swap->children[kRight] = detached->children[kRight];
446  swap->weight = detached->weight;
447  }

1930  }

1938  }

1942  }
