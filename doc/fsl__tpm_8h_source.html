517 static inline uint32_t TPM_GetStatusFlags(TPM_Type *base)
518 {
519  uint32_t statusFlags = 0;
520 
521 #if defined(FSL_FEATURE_TPM_HAS_NO_STATUS) && FSL_FEATURE_TPM_HAS_NO_STATUS
522  uint8_t chanlNumber= 0;
523 
524  /* Check the timer flag */
525  if (base->SC & TPM_SC_TOF_MASK)
526  {
527  statusFlags |= kTPM_TimeOverflowFlag;
528  }
529 
530  for(chanlNumber = 0; chanlNumber < FSL_FEATURE_TPM_CHANNEL_COUNTn(base); chanlNumber++)
531  {
532  /* Check the channel flag */
533  if(base->CONTROLS[chanlNumber].CnSC & TPM_CnSC_CHF_MASK)
534  {
535  statusFlags |= (1U << chanlNumber);
536  }
537  }
538 #else
539  statusFlags = base->STATUS;
540 #endif
541 
542  return statusFlags;
543 }

552 static inline void TPM_ClearStatusFlags(TPM_Type *base, uint32_t mask)
553 {
554 #if defined(FSL_FEATURE_TPM_HAS_NO_STATUS) && FSL_FEATURE_TPM_HAS_NO_STATUS
555  uint32_t chnlStatusFlags = (mask & 0xFF);
556  uint8_t chnlNumber = 0;
557 
558  /* Clear the timer overflow flag by writing a 0 to the bit while it is set */
559  if (mask & kTPM_TimeOverflowFlag)
560  {
561  base->SC &= ~TPM_SC_TOF_MASK;
562  }
563  /* Clear the channel flag */
564  while (chnlStatusFlags)
565  {
566  if (chnlStatusFlags & 0x1)
567  {
568  base->CONTROLS[chnlNumber].CnSC &= ~TPM_CnSC_CHF_MASK;
569  }
570  chnlNumber++;
571  chnlStatusFlags = chnlStatusFlags >> 1U;
572  }
573 #else
574  /* Clear the status flags */
575  base->STATUS = mask;
576 #endif
577 }

635 {

638  reg &= ~(TPM_SC_CLKS_MASK);
