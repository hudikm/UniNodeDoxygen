204  template <typename TIterator>
205  void assign(TIterator first, TIterator last)
206  {
207 #if defined(ETL_DEBUG)
208  difference_type d = std::distance(first, last);
209  ETL_ASSERT(d <= difference_type(capacity()), ETL_ERROR(flat_multiset_full));

228 
229  ETL_ASSERT(!full(), ETL_ERROR(flat_multiset_full));
230 
231  iterator i_element = std::lower_bound(begin(), end(), value, TKeyCompare());
232 
233  value_type* pvalue = storage.allocate<value_type>();
234  ::new (pvalue) value_type(value);
