436  template <typename TIterator>
437  void assign(TIterator first, TIterator last)
438  {
439 #if defined(ETL_DEBUG)
440  difference_type d = std::distance(first, last);
441  ETL_ASSERT(d <= difference_type(capacity()), ETL_ERROR(flat_multiset_full));
442 #endif

444  clear();
445 
446  while (first != last)
447  {
448  insert(*first++);
449  }

452  //*********************************************************************
456  //*********************************************************************
457  std::pair<iterator, bool> insert(value_type& value)
458  {
459  std::pair<iterator, bool> result(end(), false);
460 
461  ETL_ASSERT(!lookup.full(), ETL_ERROR(flat_multiset_full));
462 
463  iterator i_element = std::lower_bound(begin(), end(), value, TKeyCompare());
464 
465  if (i_element == end())
466  {
467  // At the end. Doesn't exist.
468  lookup.push_back(&value);
469  result.first = --end();
470  result.second = true;
471  }
472  else
473  {
474  // Not at the end.
475  lookup.insert(i_element.ilookup, &value);
476  result.first = i_element;
477  result.second = true;
478  }
479 
480  return result;
481  }
