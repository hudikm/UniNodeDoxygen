190  {
191  if (is_trivial_list())
192  {
193  return;
194  }
195 
196  node_t* p_last = &start_node;
197  node_t* p_current = p_last->next;
198  node_t* p_next = p_current->next;
199 
200  p_current->next = nullptr;
201 
202  while (p_next != nullptr)
203  {
204  p_last = p_current;
205  p_current = p_next;
206  p_next = p_current->next;
207 
208  p_current->next = p_last;
209  }
210 
211  join(&start_node, p_current);
212  }

729  //*************************************************************************
730  void resize(size_t n, T value)
731  {
732  ETL_ASSERT(n <= MAX_SIZE, ETL_ERROR(forward_list_full));
733 
734  size_t i = 0;
735  iterator i_node = begin();
736  iterator i_last_node;
737 
738  if (empty())
739  {
740  assign(n, value);
741  }
742  else
743  {
744  // Find where we're currently at.
745  while ((i < n) && (i_node != end()))
746  {
747  ++i;
748  i_last_node = i_node;
749  ++i_node;
750  }
751 
752  if (i_node != end())
753  {
754  // Reduce.
755  erase_after(i_last_node, end());
756  }
757  else if (i_node == end())
758  {

760  while (i < n)

763  ++i;
764  }
765  }
766  }

900  //*************************************************************************
902  //*************************************************************************
903  iterator erase_after(iterator first, iterator last)
904  {
905  if (first != end() && (first != last))
906  {
907  node_t* p_first = first.p_node;
908  node_t* p_last = last.p_node;
909  node_t* p_next = p_first->next;
910 
911  // Join the ends.
912  join(p_first, p_last);
913 
914  p_first = p_next;
915 
916  // Erase the ones in between.
917  while (p_first != p_last)

1051  //*************************************************************************
1055  //*************************************************************************
1056  template <typename TCompare>
1057  void sort(TCompare compare)
1058  {
1059  iterator p_left;
1060  iterator p_right;
1061  iterator p_node;
1062  iterator p_head;
1063  iterator p_tail;
1064  int list_size = 1;
1065  int number_of_merges;
1066  int left_size;
1067  int right_size;
1068 
1069  if (is_trivial_list())
1070  {
1071  return;
1072  }
1073 
1074  while (true)
1075  {
1076  p_left = begin();
1077  p_head = before_begin();
1078  p_tail = before_begin();
1079 
1080  number_of_merges = 0; // Count the number of merges we do in this pass.
1081 
1082  while (p_left != end())
1083  {
1084  ++number_of_merges; // There exists a merge to be done.
1085  p_right = p_left;
1086  left_size = 0;
1087 
1088  // Step 'list_size' places along from left
1089  for (int i = 0; i < list_size; ++i)
1090  {
1091  ++left_size;
1092 
1093  ++p_right;
1094 
1095  if (p_right == end())
1096  {
1097  break;
1098  }
1099  }
1100 
1101  // If right hasn't fallen off end, we have two lists to merge.
1102  right_size = list_size;
1103 
1104  // Now we have two lists. Merge them.
1105  while (left_size > 0 || (right_size > 0 && p_right != end()))
1106  {
1107  // Decide whether the next node of merge comes from left or right.
1108  if (left_size == 0)
1109  {
1110  // Left is empty. The node must come from right.
1111  p_node = p_right;
1112  ++p_right;
1113  --right_size;
1114  }
1115  else if (right_size == 0 || p_right == end())
1116  {
1117  // Right is empty. The node must come from left.
1118  p_node = p_left;
1119  ++p_left;
1120  --left_size;
1121  }
1122  else if (!compare(*p_right, *p_left))
1123  {
1124  // First node of left is lower or same. The node must come from left.
1125  p_node = p_left;
1126  ++p_left;
1127  --left_size;
1128  }
1129  else
1130  {
1131  // First node of right is lower. The node must come from right.
1132  p_node = p_right;
1133  ++p_right;
1134  --right_size;
1135  }
1136 
1137  // Add the next node to the merged head.
1138  if (p_head == before_begin())
1139  {
1140  join(p_head.p_node, p_node.p_node);
1141  p_head = p_node;
1142  p_tail = p_node;
1143  }
1144  else
1145  {
1146  join(p_tail.p_node, p_node.p_node);
1147  p_tail = p_node;
1148  }
1149 
1150  p_tail.p_node->next = nullptr;
1151  }
1152 
1153  // Now left has stepped `list_size' places along, and right has too.
1154  p_left = p_right;
1155  }
1156 
1157  // If we have done only one merge, we're finished.
1158  if (number_of_merges <= 1) // Allow for number_of_merges == 0, the empty head case
1159  {
1160  return;
1161  }
1162 
1163  // Otherwise repeat, merging lists twice the size
1164  list_size *= 2;
1165  }
1166  }
