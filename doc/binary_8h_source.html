237  template <typename TReturn, const size_t NBITS, typename TValue>
238  TReturn fold_bits(TValue value)
239  {
240  STATIC_ASSERT(integral_limits<TReturn>::bits >= NBITS, "Return type too small to hold result");
241 
242  const TValue mask = etl::power<2, NBITS>::value - 1;
243  const size_t shift = NBITS;
244 
245  // Fold the value down to fit the width.
246  TReturn folded_value = 0;
247 
248  // Keep shifting down and XORing the lower bits.
249  while (value >= etl::max_value_for_nbits<NBITS>::value)
250  {
251  folded_value ^= value & mask;
252  value >>= shift;
253  }
254 
255  // Fold the remaining bits.
256  folded_value ^= value & mask;
257 
258  return folded_value;
259  }

265  template <typename TReturn, const size_t NBITS, typename TValue>
266  TReturn sign_extend(TValue value)
267  {
268  STATIC_ASSERT(etl::is_integral<TValue>::value, "TValue not an integral type");
269  STATIC_ASSERT(etl::is_integral<TReturn>::value, "TReturn not an integral type");
270  STATIC_ASSERT(etl::is_signed<TReturn>::value, "TReturn not a signed type");
271  STATIC_ASSERT(NBITS <= std::numeric_limits<TReturn>::digits, "NBITS too large for return type");
272 
273  const TReturn negative = (TReturn(1) << (NBITS - 1));
274  TReturn signed_value = value & ((1 << NBITS) - 1);
275 
276  if ((signed_value & negative) != 0)
277  {
278  const TReturn sign_bits = ~((TReturn(1) << NBITS) - 1);
279  signed_value |= sign_bits;
280  }
281 
282  return signed_value;
283  }
