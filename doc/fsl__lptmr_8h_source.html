203 static inline void LPTMR_EnableInterrupts(LPTMR_Type *base, uint32_t mask)
204 {
205  uint32_t reg = base->CSR;
206 
207  /* Clear the TCF bit so that we don't clear this w1c bit when writing back */
208  reg &= ~(LPTMR_CSR_TCF_MASK);
209  reg |= mask;
210  base->CSR = reg;
211 }

224  /* Clear the TCF bit so that we don't clear this w1c bit when writing back */
225  reg &= ~(LPTMR_CSR_TCF_MASK);

333 static inline uint32_t LPTMR_GetCurrentTimerCount(LPTMR_Type *base)
334 {
335  /* Must first write any value to the CNR. This synchronizes and registers the current value
336  * of the CNR into a temporary register which can then be read
337  */
338  base->CNR = 0U;
339  return (uint32_t)((base->CNR & LPTMR_CNR_COUNTER_MASK) >> LPTMR_CNR_COUNTER_SHIFT);

359 static inline void LPTMR_StartTimer(LPTMR_Type *base)
360 {
361  uint32_t reg = base->CSR;
362 
363  /* Clear the TCF bit to avoid clearing the w1c bit when writing back. */
364  reg &= ~(LPTMR_CSR_TCF_MASK);
365  reg |= LPTMR_CSR_TEN_MASK;
366  base->CSR = reg;
367 }
