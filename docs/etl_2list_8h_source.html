174  void reverse()
175  {
176  if (is_trivial_list())
177  {
178  return;
179  }
180 
181  node_t* p_node = terminal_node.next;
182 
183  while (p_node != &terminal_node)
184  {
185  node_t* p_temp = p_node->previous;
186  p_node->previous = p_node->next;
187  p_node->next = p_temp;
188  p_node = p_node->previous;
189  }
190 
191  // Terminal node.
192  node_t* p_temp = p_node->previous;
193  p_node->previous = p_node->next;
194  p_node->next = p_temp;
195  }

718  template <typename TIterator>
719  void assign(TIterator first, TIterator last)
720  {
721 #if defined(ETL_DEBUG)
722  difference_type d = std::distance(first, last);
723  ETL_ASSERT(d >= 0, ETL_ERROR(list_iterator));
724  ETL_ASSERT(size_t(d) <= MAX_SIZE, ETL_ERROR(list_full));
725 #endif
726  initialise();
727 
728  // Add all of the elements.
729  while (first != last)
730  {
731  data_node_t& node = allocate_data_node(*first);
732  join(get_tail(), node);
733  join(node, terminal_node);
734  ++first;
735  }
736  }

1324  int left_size;

1327  if (is_trivial_list())
1328  {
1329  return;
1330  }
1331 
1332  while (true)
1333  {
1334  i_left = begin();
1335  i_head = end();
1336  i_tail = end();
1337 
1338  number_of_merges = 0; // Count the number of merges we do in this pass.
1339 
1340  while (i_left != end())
1341  {
1342  ++number_of_merges; // There exists a merge to be done.
1343  i_right = i_left;
1344  left_size = 0;
1345 
1346  // Step 'list_size' places along from left
1347  for (int i = 0; i < list_size; ++i)
1348  {
1349  ++left_size;
1350  ++i_right;
1351 
1352  if (i_right == end())
1353  {
1354  break;
1355  }
1356  }
1357 
1358  // If right hasn't fallen off end, we have two lists to merge.
1359  right_size = list_size;
1360 
1361  // Now we have two lists. Merge them.
1362  while (left_size > 0 || (right_size > 0 && i_right != end()))
1363  {
1364  // Decide whether the next node of merge comes from left or right.
1365  if (left_size == 0)
1366  {
1367  // Left is empty. The node must come from right.
1368  i_node = i_right++;
1369  --right_size;
1370  }
1371  else if (right_size == 0 || i_right == end())
1372  {
1373  // Right is empty. The node must come from left.
1374  i_node = i_left++;
1375  --left_size;
1376  }
1377  else if (!compare(*i_right, *i_left))
1378  {
1379  // First node of left is lower or same. The node must come from left.
1380  i_node = i_left++;
1381  --left_size;
1382  }
1383  else
1384  {
1385  // First node of right is lower. The node must come from right.
1386  i_node = i_right;
1387  ++i_right;
1388  --right_size;
1389  }
1390 
1391  // Add the next node to the merged head.
1392  if (i_head == end())
1393  {
1394  join(*i_head.p_node, *i_node.p_node);
1395  i_head = i_node;
1396  i_tail = i_node;
1397  }
1398  else
1399  {
1400  join(*i_tail.p_node, *i_node.p_node);

1402  }
1403 
1404  join(*i_tail.p_node, terminal_node);
1405  }

1408  i_left = i_right;

1412  if (number_of_merges <= 1) // Allow for number_of_merges == 0, the empty head case
