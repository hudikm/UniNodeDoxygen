2094 /*
2095  * THIS FUNCTION MUST NOT BE USED FROM APPLICATION CODE. IT IS ONLY
2096  * INTENDED FOR USE WHEN IMPLEMENTING A PORT OF THE SCHEDULER AND IS
2097  * AN INTERFACE WHICH IS FOR THE EXCLUSIVE USE OF THE SCHEDULER.
2098  *
2099  * Called from the real time kernel tick (either preemptive or cooperative),
2100  * this increments the tick count and checks if any tasks that are blocked
2101  * for a finite period required removing from a blocked list and placing on
2102  * a ready list. If a non-zero value is returned then a context switch is
2103  * required because either:
2104  * + A task was removed from a blocked list because its timeout had expired,
2105  * or
2106  * + Time slicing is in use and there is a task of equal priority to the
2107  * currently running task.
2108  */
2109 BaseType_t xTaskIncrementTick( void ) PRIVILEGED_FUNCTION;

2111 /*
2112  * THIS FUNCTION MUST NOT BE USED FROM APPLICATION CODE. IT IS AN
2113  * INTERFACE WHICH IS FOR THE EXCLUSIVE USE OF THE SCHEDULER.
2114  *
2115  * THIS FUNCTION MUST BE CALLED WITH INTERRUPTS DISABLED.
2116  *
2117  * Removes the calling task from the ready list and places it both
2118  * on the list of tasks waiting for a particular event, and the
2119  * list of delayed tasks. The task will be removed from both lists
2120  * and replaced on the ready list should either the event occur (and
2121  * there be no higher priority tasks waiting on the same event) or
2122  * the delay period expires.
2123  *
2124  * The 'unordered' version replaces the event list item value with the
2125  * xItemValue value, and inserts the list item at the end of the list.
2126  *
2127  * The 'ordered' version uses the existing event list item value (which is the
2128  * owning tasks priority) to insert the list item into the event list is task
2129  * priority order.
2130  *
2131  * @param pxEventList The list containing tasks that are blocked waiting
2132  * for the event to occur.
2133  *
2134  * @param xItemValue The item value to use for the event list item when the
2135  * event list is not ordered by task priority.
2136  *
2137  * @param xTicksToWait The maximum amount of time that the task should wait
2138  * for the event to occur. This is specified in kernel ticks,the constant
2139  * portTICK_PERIOD_MS can be used to convert kernel ticks into a real time
2140  * period.
2141  */
2142 void vTaskPlaceOnEventList( List_t * const pxEventList, const TickType_t xTicksToWait ) PRIVILEGED_FUNCTION;
2143 void vTaskPlaceOnUnorderedEventList( List_t * pxEventList, const TickType_t xItemValue, const TickType_t xTicksToWait ) PRIVILEGED_FUNCTION;
