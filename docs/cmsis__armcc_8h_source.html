517 __attribute__((always_inline)) __STATIC_INLINE uint32_t __RBIT(uint32_t value)
518 {
519  uint32_t result;
520  int32_t s = (4 /*sizeof(v)*/ * 8) - 1; /* extra shift needed at end */
521 
522  result = value; /* r will be reversed bits of v; first get LSB of v */
523  for (value >>= 1U; value; value >>= 1U)
524  {
525  result <<= 1U;
526  result |= value & 1U;
527  s--;
528  }
529  result <<= s; /* shift when v's highest bits are zero */
530  return(result);
531 }
