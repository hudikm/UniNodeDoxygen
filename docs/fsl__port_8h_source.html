291 static inline void PORT_SetMultiplePinsConfig(PORT_Type *base, uint32_t mask, const port_pin_config_t *config)
292 {
293  assert(config);
294 
295  uint16_t pcrl = *((const uint16_t *)config);
296 
297  if (mask & 0xffffU)
298  {
299  base->GPCLR = ((mask & 0xffffU) << 16) | pcrl;
300  }
301  if (mask >> 16)
302  {
303  base->GPCHR = (mask & 0xffff0000U) | pcrl;
304  }
