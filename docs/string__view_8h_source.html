605  {

608  return npos;
609  }
610 
611  position = std::min(position, size() - 1);
612 
613  const_reverse_iterator it = rbegin() + size() - position - 1;
614 
615  while (it != rend())
616  {

618 
619  for (size_t j = 0; j < viewlength; ++j)
620  {
621  if (mbegin[position] == view[j])
622  {
623  return position;
624  }
625  }
626 
627  ++it;
628  --position;
629  }
630 
631  return npos;
632  }
