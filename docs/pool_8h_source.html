353  //*************************************************************************
355  //*************************************************************************
356  char* allocate_item()
357  {
358  char* p_value = nullptr;
359 
360  // Any free space left?
361  if (items_allocated < MAX_SIZE)
362  {
363  // Initialise another one if necessary.
364  if (items_initialised < MAX_SIZE)
365  {
366  uintptr_t p = reinterpret_cast<uintptr_t>(p_buffer + (items_initialised * ITEM_SIZE));
367  *reinterpret_cast<uintptr_t*>(p) = p + ITEM_SIZE;
368  ++items_initialised;
369  }
370 
371  // Get the address of new allocated item.
372  p_value = p_next;
373 
374  ++items_allocated;
375  if (items_allocated != MAX_SIZE)
376  {
377  // Set up the pointer to the next free item
378  p_next = *reinterpret_cast<char**>(p_next);
379  }
380  else
381  {
382  // No more left!
383  p_next = nullptr;
384  }
385  }
386  else
387  {
388  ETL_ASSERT(false, ETL_ERROR(pool_no_allocation));
389  }
390 
391  return p_value;
392  }

394  //*************************************************************************
396  //*************************************************************************
397  void release_item(char* p_value)
398  {
399  // Does it belong to us?
400  ETL_ASSERT(is_item_in_pool(p_value), ETL_ERROR(pool_object_not_in_pool));
401 
402  if (p_next != nullptr)
403  {
404  // Point it to the current free item.
405  *(uintptr_t*)p_value = reinterpret_cast<uintptr_t>(p_next);
406  }
407  else
408  {
409  // This is the only free item.
410  *((uintptr_t*)p_value) = 0;
411  }
412 
413  p_next = p_value;
414 
415  --items_allocated;
416  }

418  //*************************************************************************
420  //*************************************************************************
421  bool is_item_in_pool(const char* p) const
422  {
423  // Within the range of the buffer?
424  intptr_t distance = p - p_buffer;
425  bool is_within_range = (distance >= 0) && (distance <= intptr_t((ITEM_SIZE * MAX_SIZE) - ITEM_SIZE));
426 
427  // Modulus and division can be slow on some architectures, so only do this in debug.
428 #if defined(ETL_DEBUG)
429  // Is the address on a valid object boundary?
430  bool is_valid_address = ((distance % ITEM_SIZE) == 0);
431 #else
432  bool is_valid_address = true;
433 #endif
434 
435  return is_within_range && is_valid_address;
436  }
